<body style="font-family: Arial; text-align:center; background:#f5f6fa;">

  <h1 style="background:#0077cc; color:white; padding:15px;">Student Result Management System</h1>

  <h2>Check Your Result</h2>

  <form id="resultForm">

    <input type="text" id="roll" placeholder="Enter Roll Number" required />

    <button type="submit">Get Result</button>

  </form>


  <div id="resultContainer" style="display:none; margin-top:20px;">

    <h3>Result Details</h3>

    <p><strong>Name:</strong> <span id="studentName"></span></p>

    <p><strong>Roll No:</strong> <span id="studentRoll"></span></p>

    <p><strong>Marks:</strong> <span id="studentMarks"></span></p>

    <p><strong>Grade:</strong> <span id="studentGrade"></span></p>

  </div>


  <script>

    const studentData = {

      '101': { name: 'Alice Johnson', marks: 89 },

      '102': { name: 'Rahul Sharma', marks: 76 },

      '103': { name: 'Fatima Khan', marks: 92 }

    };


    function getGrade(marks) {

      if (marks >= 90) return 'A+';

      else if (marks >= 80) return 'A';

      else if (marks >= 70) return 'B';

      else if (marks >= 60) return 'C';

      else if (marks >= 50) return 'D';

      else return 'F';

    }


    document.getElementById('resultForm').addEventListener('submit', function(e) {

      e.preventDefault();

      const roll = document.getElementById('roll').value.trim();

      const resultDiv = document.getElementById('resultContainer');


      if (studentData[roll]) {

        const student = studentData[roll];

        document.getElementById('studentName').textContent = student.name;

        document.getElementById('studentRoll').textContent = roll;

        document.getElementById('studentMarks').textContent = student.marks;

        document.getElementById('studentGrade').textContent = getGrade(student.marks);

        resultDiv.style.display = 'block';

      } else {

        alert('No result found for Roll Number ' + roll);

        resultDiv.style.display = 'none';

      }

    });

  </script>
